SELECT operator.First_name,type.type_description,contact.first_name,incident.incident_id,incident.status_id,incident.Solution_id,incident.Contact_id,incident.Operator_id,incident.Date_time,incident.Description,incident.type_id,incident_status.Description FROM incident LEFT JOIN incident_status ON incident.status_id = incident_status.Status_id LEFT JOIN operator ON incident.Operator_id = operator.Operator_id JOIN contact ON Incident.Contact_id = contact.Contact_id LEFT JOIN type ON Incident.type_id = type.type_id
